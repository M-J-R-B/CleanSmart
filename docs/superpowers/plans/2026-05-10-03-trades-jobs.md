# CleanSmart Trades & Jobs Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Seed the trade catalog, let clients post jobs (with photos and geo), and serve workers a feed filtered by trade + service radius via PostGIS.

**Architecture:** `Job` rows hold a `geography(Point)` column populated alongside lat/lng. Photos go to S3 via pre-signed PUT URLs from the API. Worker feed uses `ST_DWithin` joining `worker_profiles.home_geog` against `jobs.location_geog`. Trade catalog is a seeded lookup table.

**Tech Stack:** AWS S3 SDK v3, PostGIS, Prisma, Fastify, zod.

**Depends on:** Plans 1-2 complete and merged.

```ts
import { S3Client, PutObjectCommand } from "@aws-sdk/client-s3";
import { getSignedUrl } from "@aws-sdk/s3-request-presigner";

export interface StorageConfig {
  region: string;
  bucket: string;
  accessKeyId: string;
  secretAccessKey: string;
  endpoint?: string;
  publicBaseUrl?: string;
}

export interface StorageService {
  presignPut(key: string, contentType: string): Promise<{ url: string; key: string }>;
  publicUrl(key: string): string;
}

export function createStorageService(cfg: StorageConfig): StorageService {
  const client = new S3Client({
    region: cfg.region,
    credentials: { accessKeyId: cfg.accessKeyId, secretAccessKey: cfg.secretAccessKey },
    endpoint: cfg.endpoint,
    forcePathStyle: !!cfg.endpoint,
  });
  return {
    async presignPut(key, contentType) {
      const cmd = new PutObjectCommand({ Bucket: cfg.bucket, Key: key, ContentType: contentType });
      const url = await getSignedUrl(client, cmd, { expiresIn: 600 });
      return { url, key };
    },
    publicUrl(key) {
      if (cfg.publicBaseUrl) return `${cfg.publicBaseUrl.replace(/\/$/, "")}/${key}`;
      return `https://${cfg.bucket}.s3.${cfg.region}.amazonaws.com/${key}`;
    },
  };
}
```

```ts
export const JobUrgencySchema = z.enum(["now", "this_week", "scheduled"]);
export type JobUrgencyT = z.infer<typeof JobUrgencySchema>;

export const CreateJobSchema = z
  .object({
    tradeId: z.string().uuid(),
    title: z.string().min(3).max(120),
    description: z.string().min(10).max(4000),
    address: z.string().min(5).max(300),
    lat: z.number().gte(-90).lte(90),
    lng: z.number().gte(-180).lte(180),
    urgency: JobUrgencySchema,
    scheduledAt: z.string().datetime().optional(),
    budgetMinCents: z.number().int().nonnegative().optional(),
    budgetMaxCents: z.number().int().nonnegative().optional(),
    photoKeys: z.array(z.string().min(1)).max(10).default([]),
  })
  .refine(
    (v) => v.urgency !== "scheduled" || !!v.scheduledAt,
    { message: "scheduledAt required for scheduled urgency" },
  )
  .refine(
    (v) => v.budgetMinCents == null || v.budgetMaxCents == null || v.budgetMinCents <= v.budgetMaxCents,
    { message: "budgetMin must be <= budgetMax" },
  );
export type CreateJob = z.infer<typeof CreateJobSchema>;

export const PresignPhotoSchema = z.object({
  contentType: z.enum(["image/jpeg", "image/png", "image/webp"]),
});
export type PresignPhoto = z.infer<typeof PresignPhotoSchema>;

export const FeedQuerySchema = z.object({
  cursor: z.string().optional(),
  limit: z.coerce.number().int().min(1).max(50).default(20),
  urgency: z.array(JobUrgencySchema).optional(),
});
```

```ts
import { randomUUID } from "node:crypto";
import type { FastifyInstance } from "fastify";
import { CreateJobSchema, PresignPhotoSchema, FeedQuerySchema } from "@cleansmart/shared";
import { requireAuth, requireRole } from "../middleware/auth";

export async function jobRoutes(app: FastifyInstance) {
  const { prisma, storage } = app.container;

  app.post("/jobs/photo-url", { preHandler: [requireAuth, requireRole("client")] }, async (req) => {
    const body = PresignPhotoSchema.parse(req.body);
    const ext = body.contentType === "image/jpeg" ? "jpg" : body.contentType.split("/")[1];
    const key = `jobs/uploads/${req.userId}/${randomUUID()}.${ext}`;
    return storage.presignPut(key, body.contentType);
  });

  app.post("/jobs", { preHandler: [requireAuth, requireRole("client")] }, async (req, reply) => {
    const body = CreateJobSchema.parse(req.body);
    const job = await prisma.job.create({
      data: {
        clientId: req.userId!,
        tradeId: body.tradeId,
        title: body.title,
        description: body.description,
        address: body.address,
        lat: body.lat,
        lng: body.lng,
        urgency: body.urgency,
        scheduledAt: body.scheduledAt ? new Date(body.scheduledAt) : null,
        budgetMinCents: body.budgetMinCents,
        budgetMaxCents: body.budgetMaxCents,
        photos: body.photoKeys.length ? { createMany: { data: body.photoKeys.map((s3Key) => ({ s3Key })) } } : undefined,
      },
    });
    await prisma.$executeRaw`
      UPDATE jobs SET location_geog = ST_SetSRID(ST_MakePoint(${body.lng}, ${body.lat}), 4326)::geography
      WHERE id = ${job.id}::uuid
    `;
    reply.status(201);
    return job;
  });

  app.get("/jobs/:id", { preHandler: requireAuth }, async (req, reply) => {
    const id = (req.params as { id: string }).id;
    const job = await prisma.job.findUnique({
      where: { id },
      include: { trade: true, photos: true, client: { select: { id: true, name: true, avatarUrl: true } } },
    });
    if (!job) return reply.status(404).send({ error: { code: "not_found", message: "Job not found" } });
    return job;
  });

  app.delete("/jobs/:id", { preHandler: [requireAuth, requireRole("client")] }, async (req, reply) => {
    const id = (req.params as { id: string }).id;
    const job = await prisma.job.findUnique({ where: { id } });
    if (!job) return reply.status(404).send({ error: { code: "not_found", message: "Job not found" } });
    if (job.clientId !== req.userId) return reply.status(403).send({ error: { code: "forbidden", message: "Not your job" } });
    if (job.status !== "open") return reply.status(409).send({ error: { code: "invalid_state", message: "Cannot cancel after acceptance" } });
    await prisma.job.update({ where: { id }, data: { status: "cancelled" } });
    reply.status(204).send();
  });
}
```

```ts
app.get("/jobs/feed", { preHandler: [requireAuth, requireRole("worker")] }, async (req, reply) => {
  const q = FeedQuerySchema.parse(req.query);
  const worker = await prisma.workerProfile.findUnique({
    where: { userId: req.userId! },
    include: { trades: true },
  });
  if (!worker || !worker.homeLat || !worker.homeLng) {
    return reply.status(409).send({ error: { code: "profile_incomplete", message: "Set home location first" } });
  }
  const tradeIds = worker.trades.map((t) => t.tradeId);
  if (tradeIds.length === 0) return { items: [], nextCursor: null };

  const radiusMeters = worker.serviceRadiusKm * 1000;
  const urgencyFilter = q.urgency?.length ? q.urgency : ["now", "this_week", "scheduled"];
  const cursorClause = q.cursor ? `AND j.created_at < (SELECT created_at FROM jobs WHERE id = $4::uuid)` : "";
  const params: any[] = [tradeIds, urgencyFilter, radiusMeters];
  if (q.cursor) params.push(q.cursor);

  const sql = `
    SELECT j.id, j.title, j.description, j.urgency, j.created_at, j.lat, j.lng, j.address,
           j.budget_min_cents, j.budget_max_cents,
           ST_Distance(j.location_geog, w.home_geog) AS distance_m
    FROM jobs j
    JOIN worker_profiles w ON w.user_id = $${params.length + 1}::uuid
    WHERE j.status = 'open'
      AND j.trade_id = ANY($1::uuid[])
      AND j.urgency::text = ANY($2::text[])
      AND ST_DWithin(j.location_geog, w.home_geog, $3)
      ${cursorClause}
    ORDER BY (j.urgency = 'now') DESC, (j.urgency = 'this_week') DESC, j.created_at DESC
    LIMIT ${q.limit + 1}
  `;
  params.push(req.userId!);
  const rows = await prisma.$queryRawUnsafe<any[]>(sql, ...params);
  const items = rows.slice(0, q.limit);
  const nextCursor = rows.length > q.limit ? items[items.length - 1].id : null;
  return { items, nextCursor };
});
```
